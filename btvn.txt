#include<iostream>

class songuyen{
	public:

    cout<<"nhap vao so phan tu :";

	for(int i=0;i<n;i++){
		cout<<"a["<<i<<"]=";
		cin>>a[i];
	}
}
void xuat(){
	cout<<"\ncac so vua nhap la :\n";
	for(int i=0;i<n;i++){

void xuatnguoc(){
	cout<<"xuat nguoc cac so vua nhap :\n";
	for(int i=n-1;i>=0;i--){
		cout<<a[i]<<" ";
	}
	cout<<endl;
}
void tim(){
	cout<<"nhap vao so can tim : ";
	cin>>k;
	for(int i=0;i<n;i++){
		if(a[i]==k)
		cout<<"\ntim thay so o vi tri :"<<"a["<<i<<"] ";
	}
	cout<<endl;
}	

void tangdan(){
	cout<<"\nsap xep theo thu tu tang dan\n";
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
		if(a[i]>a[j])
		swap(a[i],a[j]);
        	}
